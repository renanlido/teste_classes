# LaneFlow (Eclusa) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Construir a máquina de estados (State Pattern) de uma eclusa de acesso veicular com 3 cancelas, orquestrando integrações emuladas em memória (cancela/CLP, ALPR, facial, backend Recintos/SEV, EventBus).

**Architecture:** Domínio puro (`Operation`, `Cancela`, serviços), `LaneFlow` com State Pattern (estados estendem `LaneStateBase`), integrações por interface + emulação em memória lado a lado. `Lane` é singleton por id via `LaneRegistry`. `Operation` é o contexto que amarra a passagem. Sem DI/use-cases/repository abstratos — foco na lógica.

**Tech Stack:** TypeScript (ESM, strict), `tsx`, testes com `node:test` + `node:assert/strict` (zero dependência de teste).

**Refinamento sobre o spec:** o contrato `LaneState.handle` retorna `LaneState | void` (o estado escolhe o próximo; `LaneFlow` faz `transitionTo`). Eventos de dados passivos (`placaLida`/`pessoaDetectada`/`pesoMedido`) são gravados na `Operation` por `LaneFlow.dispatch` antes de delegar ao estado.

import { randomUUID } from "node:crypto";
import type { Lado, Plate, Pessoa, Agendamento, SevResult } from "./types.js";

export class Operation {
  readonly id: string;
  readonly lado: Lado;
  placas: Plate[] = [];
  pessoa: Pessoa | null = null;
  pesado = false;
  agendamento: Agendamento | null = null;
  sev: SevResult | null = null;

  private readonly startTime: Date;
  private endTime?: Date;

  constructor(lado: Lado) {
    this.id = randomUUID();
    this.lado = lado;
    this.startTime = new Date();
  }

  get placa(): Plate | undefined {
    return this.placas[0];
  }

  endOperation(): void {
    this.endTime = new Date();
  }

  operationTime(): number {
    if (!this.endTime) {
      throw new Error("Operação não terminou");
    }
    return this.endTime.getTime() - this.startTime.getTime();
  }
}
```

import { randomUUID } from "node:crypto";
import type { CommandGate } from "../integrations/CommandGate.js";

const MAX_TENTATIVAS = 3;

export abstract class CancelaBase {
  protected readonly id: string;
  protected estadoCancela: "aberto" | "fechado" = "fechado";

  constructor(protected readonly command: CommandGate) {
    this.id = randomUUID();
  }

  get estado(): "aberto" | "fechado" {
    return this.estadoCancela;
  }

  abstract abreCancela(): Promise<void>;
  abstract fechaCancela(): Promise<void>;
}

export class Cancela extends CancelaBase {
  async abreCancela(): Promise<void> {
    let tentativas = 0;
    while (this.estadoCancela !== "aberto") {
      tentativas++;
      const retorno = await this.command.abreCancela(this.id);
      if (retorno.type === "failure") {
        throw new Error(retorno.message);
      }
      this.estadoCancela = await this.command.consultaEstadoCancela(this.id);
      if (this.estadoCancela !== "aberto" && tentativas >= MAX_TENTATIVAS) {
        throw new Error("tempo limite ao abrir cancela");
      }
    }
  }

  async fechaCancela(): Promise<void> {
    let tentativas = 0;
    while (this.estadoCancela !== "fechado") {
      tentativas++;
      await this.command.fechaCancela(this.id);
      this.estadoCancela = await this.command.consultaEstadoCancela(this.id);
      if (this.estadoCancela !== "fechado" && tentativas >= MAX_TENTATIVAS) {
        throw new Error("tempo limite ao fechar cancela");
      }
    }
  }
}
```

```ts
import type { Operation } from "./Operation.js";
import type { BackendPort } from "../integrations/BackendPort.js";
import type { LaneConfig } from "../flow/LaneConfig.js";

export interface ResultadoValidacao {
  ok: boolean;
  motivo?: string;
}

function comTimeout<T>(p: Promise<T>, ms: number): Promise<T> {
  return new Promise<T>((resolve, reject) => {
    const timer = setTimeout(() => reject(new Error("timeout backend")), ms);
    p.then(
      (v) => {
        clearTimeout(timer);
        resolve(v);
      },
      (e) => {
        clearTimeout(timer);
        reject(e);
      },
    );
  });
}

export class ValidacaoService {
  async avaliar(cfg: LaneConfig, op: Operation, backend: BackendPort): Promise<ResultadoValidacao> {
    const ms = cfg.timeouts.backendMs;
    try {
      if (cfg.facialAtivo && !op.pessoa) {
        return { ok: false, motivo: "sem pessoa" };
      }
      if (op.pessoa) {
        const ag = await comTimeout(backend.agendamento(op.pessoa), ms);
        if (!ag.valido) {
          return { ok: false, motivo: "agendamento inválido" };
        }
        const noCadastro = await comTimeout(backend.placaNoCadastro(op.pessoa, op.placa), ms);
        if (!noCadastro) {
          return { ok: false, motivo: "placa fora do cadastro" };
        }
      }
      if (cfg.sevAtivo && op.pesado && op.pessoa) {
        const sev = await comTimeout(backend.sev(op.pessoa, op.placa), ms);
        if (!sev.ok) {
          return { ok: false, motivo: "sem SEV" };
        }
      }
      return { ok: true };
    } catch (e) {
      return { ok: false, motivo: e instanceof Error ? e.message : "timeout backend" };
    }
  }
}
```

```ts
import type { FlowEvent, FlowDeps } from "./events.js";
import { EVENTOS_DADOS } from "./events.js";
import type { LaneConfig } from "./LaneConfig.js";
import type { LaneState, LaneFlowApi } from "./LaneStateBase.js";
import type { Operation } from "../domain/Operation.js";

export abstract class LaneFlowBase {
  abstract getFlow(): { estado: string; operacaoId: string | null };
  abstract getState(): string;
}

export class LaneFlow extends LaneFlowBase implements LaneFlowApi {
  operation: Operation | null = null;
  onFail: (motivo: unknown) => LaneState;

  private state: LaneState | null = null;
  private watchdog: ReturnType<typeof setTimeout> | null = null;
  private pendingFail: unknown = null;

  constructor(
    readonly cfg: LaneConfig,
    readonly deps: FlowDeps,
  ) {
    super();
    this.onFail = () => {
      throw new Error("onFail não configurado");
    };
  }

  getState(): string {
    return this.state?.nome ?? "—";
  }

  getFlow(): { estado: string; operacaoId: string | null } {
    return { estado: this.getState(), operacaoId: this.operation?.id ?? null };
  }

  async start(estadoInicial: LaneState): Promise<void> {
    await this.runOnEnter(estadoInicial);
  }

  async dispatch(ev: FlowEvent): Promise<void> {
    if ((EVENTOS_DADOS as readonly string[]).includes(ev.tipo)) {
      this.record(ev);
      return;
    }
    if (!this.state) return;
    const next = this.state.handle(ev, this);
    if (next) {
      await this.transitionTo(next);
    }
  }

  async transitionTo(novo: LaneState): Promise<void> {
    if (this.state) {
      await this.state.onExit(this);
    }
    await this.runOnEnter(novo);
  }

  fail(motivo: unknown): void {
    this.pendingFail = motivo;
  }

  armWatchdog(ms: number): void {
    this.clearWatchdog();
    this.watchdog = setTimeout(() => {
      void this.dispatch({ tipo: "timeout" });
    }, ms);
  }

  clearWatchdog(): void {
    if (this.watchdog) {
      clearTimeout(this.watchdog);
      this.watchdog = null;
    }
  }

  log(...args: unknown[]): void {
    console.log("[LaneFlow]", ...args);
  }

  private async runOnEnter(novo: LaneState): Promise<void> {
    this.state = novo;
    this.pendingFail = null;
    try {
      await novo.onEnter(this);
    } catch (e) {
      this.pendingFail = e;
    }
    if (this.pendingFail !== null) {
      const motivo = this.pendingFail;
      this.pendingFail = null;
      await this.transitionTo(this.onFail(motivo));
    }
  }

  private record(ev: FlowEvent): void {
    if (!this.operation) return;
    if (ev.tipo === "placaLida") this.operation.placas.push(ev.placa);
    else if (ev.tipo === "pessoaDetectada") this.operation.pessoa = ev.pessoa;
    else if (ev.tipo === "pesoMedido") this.operation.pesado = ev.pesado;
  }
}
```

export class InMemoryEventBus implements EventBus {
  private handlers = new Map<string, ((payload: unknown) => void)[]>();

  publish(topic: string, payload: unknown): void {
    for (const h of this.handlers.get(topic) ?? []) {
      h(payload);
    }
  }

  subscribe(topic: string, handler: (payload: unknown) => void): void {
    const list = this.handlers.get(topic) ?? [];
    list.push(handler);
    this.handlers.set(topic, list);
  }
}

import type { Lane } from "./Lane.js";

export class LaneRegistry {
  private static lanes = new Map<string, Lane>();

  static get(id: string, factory: () => Lane): Lane {
    const existente = this.lanes.get(id);
    if (existente) return existente;
    const nova = factory();
    this.lanes.set(id, nova);
    return nova;
  }

  static peek(id: string): Lane | undefined {
    return this.lanes.get(id);
  }

  static reset(): void {
    this.lanes.clear();
  }
}
